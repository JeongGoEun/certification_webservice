===> 달력의 기본 모양임다. 만드는 방법은 아래와 같슴다.

우선 html의 <head> ~ </head> 부분을 다음과 같이 작성합니다.

<HEAD>
<TITLE>Calendar</TITLE>
<meta http-equiv="Content-type" content="text/html; charset=euc-kr">
<style>
<!-- 자동으로 테이블 등의 글자 크기와 종류 지정 -->
table,tr,td,select,input,div,form, { text-decoration:none; color:black; font-size:10pt; font-family:굴림;}
<!-- 링크관련 스타일 -->
:link {TEXT-DECORATION: none}
:visited {TEXT-DECORATION: none}
:hover {TEXT-DECORATION: none}
</style>

<script src="calendar.js"></script>
</HEAD>

그리고 나서 달력을 html에 아래와 같이 추가합니다.

<script>
c = new Calendar('c', 'frm');
c.show();
</script>

===> 위의 달력을 변형해 봄니다. 아래와 같이 Calendar 객체의 인수로 날짜와 format, 요일 표시를 지정할 수 있습니다.

<script>
cal2 = new Calendar('cal2', 'frm2', '2002', '05', '07', 'yyyy-mm-dd', 3);
cal2.show();
</script>

===> 또는 아래와 같이 달력의 옵션을 수정할 수도 있습니다.

<script>
cal3 = new Calendar('cal3', 'frm3');
cal3.tfDate = 'dateField';
cal3.bgcolor = '#D1EBF3'; // 요일 배경색
cal3.bgcolor2 = '#E9F8FC'; // 내용 배경색
cal3.linecolor = '#D1EBF3'; // 라인 배경색
cal3.lineheight = '20px'; // 라인 높이
cal3.width = '300'; // 달력 너비
cal3.tfDateLineCol = 'green'; // 텍스트필드 라인색
cal3.tfCol = '#E9F8FC'; // 내용 텍스트필드 배경색
cal3.focusCol = 'yellow'; // 포커스 색
cal3.show();
</script>

===> 보다 랜덤한 칼라의 달력을 원하신다면 아래와 같이 달력 색을 랜덤하게 지정할 수 있습니다.

<script>
// 달력 보여주기
cal = new Calendar('cal', 'frm1');
cal.tfDate = 'CALDDATE'; // 날짜 입력 필드명

var gb = Math.round(Math.random()*6);
// gb - 0~6 : 핑크색, 오렌지색, 노란색, 연두색, 보라색, 녹색, 하늘색
var dayColor = new Array("#F4D7F1", "#FAD9B7", "#E3E0CC", "#E2F3CD", "#EBD8FB", "#CAF2E7", "#D1EBF3");
var dateColor = new Array("#FDF2FC", "#FBF3EA", "#FDF9E5", "#F5FEEA", "#F5EEFB", "#EFFCF5", "#E9F8FC");
var focusColor = new Array("yellow", "white", "#DDDDDD", "#DDDDDD", "white", "#DDDDDD", "#DDDDDD");

cal.bgcolor = dayColor[gb]; // 요일 바탕색
cal.bgcolor2 = dateColor[gb]; // 내용 바탕색
cal.linecolor = dayColor[gb]; // 요일 바탕 라인색
cal.lineheight = '23px'; // 요일 바탕 높이
cal.width = '180'; // 달력 너비
cal.tfDateLineCol = 'black'; // 날짜 입력 필드 라인색
cal.tfCol = dateColor[gb]; // 내용 필드 바탕색
cal.focusCol = focusColor[gb]; // 포커드 필드 바탕색
cal.show();
</script>
